Example 1:

<button type="button" class="btn btn-primary btn-copy1" data-clipboard-text="This is Example 1.">Example 1</button>

Example 2:

<button type="button" class="btn btn-primary btn-copy2">Copy2</button>
<textarea class="form-control" rows="3">This is Example 2.</textarea>

JS:

$('.btn-copy1').copy();

$('.btn-copy2').copy({
	copy: function(_this){
		return _this.next('textarea').val();
	},
	afterCopy: function(res, _this){
		if(res==true){
			alert('Copied text to clipboard.');
		}else{
			alert('Copy failed.');
		}
	}
});
