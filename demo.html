Example 2:

<button type="button" class="btn btn-primary btn-copy2">Copy2</button>
<textarea class="form-control" rows="3">This is Example 2.</textarea>

Example 3:

<button type="button" class="btn btn-primary btn-copy3" data-clipboard-text="This is Example 3.">Example 3</button>

Example 4:

<button type="button" class="btn btn-primary btn-copy4" data-clipboard-text="This is Example 4.">Example 4</button>

JS:

$('.btn-copy1').copy();

$('.btn-copy2').copy({
	copy: function(_this){
		return _this.next('textarea').val();
	},
	afterCopy: function(res, _this){
		if(res==true){
			alert('Copied text to clipboard.');
		}else{
			alert('Copy failed.');
		}
	}
});

$('.btn-copy3').copy({tooltip: false});

$('.btn-copy4').copy({
	tooltip: {
		title: 'click to copy',
		afterCopyText: 'copied!!'
	}
});
